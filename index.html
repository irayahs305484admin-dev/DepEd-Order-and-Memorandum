function doGet(e) { // Using an RSS-to-JSON proxy to bypass potential Cloudflare blocks on gov.ph and reliably parse XML const ordersApiUrl = 'https://api.rss2json.com/v1/api.json?rss_url=' + encodeURIComponent('https://www.deped.gov.ph/category/issuances/deped-orders/feed/'); const memosApiUrl = 'https://api.rss2json.com/v1/api.json?rss_url=' + encodeURIComponent('https://www.deped.gov.ph/category/issuances/deped-memoranda/feed/'); const orders = fetchFromAPI(ordersApiUrl, 'orders'); const memos = fetchFromAPI(memosApiUrl, 'memoranda'); const allData = orders.concat(memos); return ContentService.createTextOutput(JSON.stringify(allData)) .setMimeType(ContentService.MimeType.JSON); } function fetchFromAPI(apiUrl, type) { const results = []; try { const response = UrlFetchApp.fetch(apiUrl, { muteHttpExceptions: true }); if (response.getResponseCode() !== 200) return results; const data = JSON.parse(response.getContentText()); if (data.status !== 'ok') return results; for (let i = 0; i < data.items.length; i++) { const item = data.items[i]; const titleRaw = item.title; const link = item.link; const pubDate = item.pubDate; let num = "0"; // Safe cross-browser date parsing let year = new Date(pubDate.replace(/-/g, '/')).getFullYear().toString(); // Capture the exact number including leading zeros: (\d+) const titleRegex = /(?:DO|DM|Order|Memo).*?(?:No\.?)?\s*(\d+)[A-Za-z]?,\s*s\.\s*(\d{4})/i; const parsedTitle = titleRaw.match(titleRegex); if (parsedTitle) { num = parsedTitle[1]; year = parsedTitle[2]; } else { const yearMatch = titleRaw.match(/\b(20\d{2})\b/); if (yearMatch) year = yearMatch[1]; const numMatch = titleRaw.match(/\b(\d+)\b/); // fallback without stripping zeroes if (numMatch) num = numMatch[1]; } // Clean up common HTML entities generated by WordPress let cleanTitle = titleRaw.replace(/–/g, '-').replace(/’/g, "'").replace(/&/g, '&'); const dateObj = new Date(pubDate.replace(/-/g, '/')); const formattedDate = Utilities.formatDate(dateObj, "Asia/Manila", "MMM dd, yyyy"); results.push({ type: type, num: num, year: year, title: cleanTitle, date: formattedDate, url: link }); } } catch (e) { Logger.log("Error: " + e.message); } return results; }
